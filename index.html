Chrome Extension Installation

For complete ad-free experience, install our Chrome extension:

1 Create Extension Folder

Create a new folder named "youtube-music-adblocker" with the following structure:

youtube-music-adblocker/
manifest.json
background.js
content.js
icons/
icon-16.png
icon-48.png
icon-128.png
2 Create Manifest File

Create manifest.json with this content:

{ "name": "YouTube Music Ad Blocker Pro", "version": "1.0", "description": "Advanced ad blocking for YouTube Music and other sites", "permissions": [ "webRequest", "webRequestBlocking", "", "storage" ], "background": { "scripts": ["background.js"], "persistent": true }, "content_scripts": [ { "matches": ["*://*/*"], "js": ["content.js"], "run_at": "document_start" } ], "icons": { "16": "icons/icon-16.png", "48": "icons/icon-48.png", "128": "icons/icon-128.png" }, "manifest_version": 2 }
3 Create Background Script

Create background.js with this content:

// Comprehensive ad blocking filters const adFilters = [ "*://*.googlesyndication.com/*", "*://*.google-analytics.com/*", "*://*.doubleclick.net/*", "*://*.googlesyndication.com/*", "*://*.google.com/pagead/*", "*://*.googletagservices.com/*", "*://*.gstatic.com/cv/js/sender/*", "*://*.youtube.com/pagead/*", "*://*.youtube.com/api/stats/ads*", "*://*.youtube.com/ptracking*", "*://*.youtube.com/get_midroll_*", "*://*.youtube.com/pagead/*", "*://*.youtube.com/advertising*", "*://*.youtube.com/api/stats/qoe*", "*://*.quantserve.com/*", "*://*.scorecardresearch.com/*", "*://*.zedo.com/*", "*://creative.ak.fbcdn.net/*", "*://*.adbrite.com/*", "*://*.exponential.com/*", "*://*.2mdn.net/*", "*://*.aniview.com/*", "*://*.adsafeprotected.com/*", "*://*.adsystem.com/*", "*://*.adnxs.com/*", "*://*.amazon-adsystem.com/*", "*://*.facebook.com/tr/*", "*://*.twitter.com/i/ads/*" ]; // Block requests before they are made chrome.webRequest.onBeforeRequest.addListener( function(details) { console.log("üîí Blocking ad request:", details.url); return { cancel: true }; }, { urls: adFilters }, ["blocking"] ); // Remove tracking headers chrome.webRequest.onHeadersReceived.addListener( function(details) { const responseHeaders = details.responseHeaders.filter(header => !header.name.toLowerCase().includes('google') && !header.name.toLowerCase().includes('ad') && !header.name.toLowerCase().includes('track') && !header.name.toLowerCase().includes('analytics') ); return { responseHeaders }; }, { urls: ["*://*.youtube.com/*", "*://*.youtube-nocookie.com/*"] }, ["blocking", "responseHeaders"] );
4 Create Content Script

Create content.js with this content:

// Content script to interact with web pages (function() { 'use strict'; // Function to detect and remove ads from the page function removeAdsFromPage() { const adSelectors = [ '[class*="ad"]', '[id*="ad"]', '[class*="banner"]', '[id*="banner"]', '.ad-container', '.ad-wrapper', '.adsense', '.google-ad', '.advertisement', '.ad-unit', '.video-ads' ]; adSelectors.forEach(selector => { const elements = document.querySelectorAll(selector); elements.forEach(element => { if (element.innerHTML.includes('ad') || element.getAttribute('id')?.includes('ad') || element.getAttribute('class')?.includes('ad')) { element.remove(); } }); }); } // Function to block YouTube ads specifically function blockYouTubeAds() { const youtubeAdSelectors = [ '.ad-display', '.video-ads', '.ytp-ad-module', '.ad-container', '#player-ads', '.ytd-companion-slot-renderer' ]; youtubeAdSelectors.forEach(selector => { const elements = document.querySelectorAll(selector); elements.forEach(element => element.remove()); }); // Try to skip ads by simulating clicks on skip buttons const skipButtons = document.querySelectorAll('.ytp-ad-skip-button, .ytp-ad-skip-button-modern'); skipButtons.forEach(button => { if (button && button.click) { button.click(); } }); } // Initialize ad blocking function init() { console.log('üöÄ YouTube Music Ad Blocker Pro activated'); // Remove existing ads removeAdsFromPage(); blockYouTubeAds(); // Set up mutation observer to handle dynamically loaded ads const observer = new MutationObserver(function(mutations) { mutations.forEach(function(mutation) { if (mutation.addedNodes.length) { removeAdsFromPage(); blockYouTubeAds(); } }); }); // Start observing observer.observe(document.body, { childList: true, subtree: true }); // Periodic cleanup setInterval(() => { removeAdsFromPage(); blockYouTubeAds(); }, 2000); } // Wait for page to load if (document.readyState === 'loading') { document.addEventListener('DOMContentLoaded', init); } else { init(); } })();
5 Install Extension

Install the extension in Chrome:

  • Open Chrome and go to chrome://extensions/
  • Enable "Developer mode" in the top right
  • Click "Load unpacked" and select your extension folder
  • The extension will now be active and blocking ads!
